71 bool chase_work_stealing_deque<T, Policies...>::try_push(value_type item) {
72  auto b = bottom.load(std::memory_order_relaxed);
73  auto t = top.load(std::memory_order_relaxed);
74  auto size = b - t;
75  if (size >= items.capacity()) {
76  if (items.can_grow()) {
77  items.grow(b, t);
78  assert(size < items.capacity());
79  // TODO - need to update top??
80  }
81  else
82  return false;
83  }
84

86 
87  // (1) - this release-store synchronizes-with the seq-cst-load (3)
88  bottom.store(b + 1, std::memory_order_release);

92 template <class T, class... Policies>
93 bool chase_work_stealing_deque<T, Policies...>::try_pop(value_type &result) {
94  auto b = bottom.load(std::memory_order_relaxed);
95  auto t = top.load(std::memory_order_relaxed);
96  if (b == t)
97  return false;
98 
99  --b;
100  // (2) - this seq-cst-store enforces a total order with the seq-cst-load (3)
101  bottom.store(b, std::memory_order_seq_cst);
102 
103  auto item = items.get(b, std::memory_order_relaxed);
104  t = top.load(std::memory_order_relaxed);
105  if (b > t) {
106  result = item;
107  return true;
108  }
109 
110  if (b == t) {
111  if (top.compare_exchange_strong(t, t + 1, std::memory_order_relaxed)) {
112  bottom.store(t + 1, std::memory_order_relaxed);
113  result = item;
114  return true;
115  } else {
116  bottom.store(t, std::memory_order_relaxed);
117  return false;
118  }
119  }
120 
121  assert(b == t - 1);
122  bottom.store(t, std::memory_order_relaxed);
123  return false;
124 }

127 bool chase_work_stealing_deque<T, Policies...>::try_steal(value_type &result) {
128  auto t = top.load(std::memory_order_relaxed);
129 
130  // (3) - this seq-cst-load enforces a total order with the seq-cst-store (2)
131  // and synchronizes-with the release-store (1)
132  auto b = bottom.load(std::memory_order_seq_cst);
133  auto size = (int)b - (int)t;
134  if (size <= 0)
135  return false;
136 
137  auto item = items.get(t, std::memory_order_relaxed);
138  if (top.compare_exchange_strong(t, t + 1, std::memory_order_relaxed)) {
139  result = item;
140  return true;
141  }
142
