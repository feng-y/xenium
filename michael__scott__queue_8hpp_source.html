104 {
105  node* n = new node{};
106  n->value = std::move(value);
107 
108  backoff backoff;
109 
110  guard_ptr t;
111  for (;;)
112  {
113  // Get the old tail pointer.
114  // (3) - this acquire-load synchronizes-with the release-CAS (5, 7, 10)
115  t.acquire(tail, std::memory_order_acquire);
116 
117  // Help update the tail pointer if needed.
118  // (4) - this acquire-load synchronizes-with the release-CAS (6)
119  auto next = t->next.load(std::memory_order_acquire);
120  if (next.get() != nullptr)
121  {
122  marked_ptr expected(t.get());
123  // (5) - this release-CAS synchronizes-with the acquire-load (3)
124  tail.compare_exchange_weak(expected, next, std::memory_order_release, std::memory_order_relaxed);
125  continue;
126  }
127 
128  // Attempt to link in the new element.
129  marked_ptr null{};
130  // (6) - this release-CAS synchronizes-with the acquire-load (2, 4, 9).
131  if (t->next.compare_exchange_weak(null, n, std::memory_order_release, std::memory_order_relaxed))
132  break;
133 
134  backoff();
135  }
136 
137  // Swing the tail to the new element.
138  marked_ptr expected = t.get();
139  // (7) - this release-CAS synchronizes-with the acquire-load (3)
140  tail.compare_exchange_strong(expected, n, std::memory_order_release, std::memory_order_relaxed);

144 bool michael_scott_queue<T, Policies...>::try_pop(T &result)
145 {
146  backoff backoff;
147 
148  guard_ptr h;
149  for (;;)
150  {
151  // Get the old head and tail elements.
152  // (8) - this acquire-load synchronizes-with the release-CAS (11)
153  h.acquire(head, std::memory_order_acquire);
154 
155  // Get the head element's successor.
156  // (9) - this acquire-load synchronizes-with the release-CAS (6).
157  auto next = acquire_guard(h->next, std::memory_order_acquire);
158  if (head.load(std::memory_order_relaxed).get() != h.get())
159  continue;
160 
161  // If the head (dummy) element is the only one, return false to signal that
162  // the operation has failed (no element has been returned).
163  if (next.get() == nullptr)
164  return false;
165 
166  marked_ptr t = tail.load(std::memory_order_relaxed);
167 
168  // There are multiple elements. Help update tail if needed.
169  if (h.get() == t.get())
170  {
171  // (10) - this release-CAS synchronizes-with the acquire-load (3)
172  tail.compare_exchange_weak(t, next, std::memory_order_release, std::memory_order_relaxed);
173  continue;
174  }
175 
176  // Attempt to update the head pointer so that it points to the new dummy node.
177  marked_ptr expected(h.get());
178  // (11) - this release-CAS synchronizes-with the acquire-load (1, 8)
179  if (head.compare_exchange_weak(expected, next, std::memory_order_release, std::memory_order_relaxed))
180  {
181  // return the data of head's successor; it is the new dummy node.
182  result = std::move(next->value);
183  break;
184  }
185 
186  backoff();
187  }
188 
189  // The old dummy node has been unlinked, so reclaim it.
190  h.reclaim();
191 
192  return true;
193 }
